5   import java.io.InputStream;

7   import org.junit.Assert;
8   import org.junit.Test;
9   
10  import co.nstant.in.cbor.model.DataItem;
11  import co.nstant.in.cbor.model.HalfPrecisionFloat;
12  
13  public abstract class AbstractHalfPrecisionFloatTest {
14  
15      private final float value;
16      private final byte[] encodedValue;
17  
18      public AbstractHalfPrecisionFloatTest(float value, byte[] encodedValue) {
19          this.value = value;
20          this.encodedValue = encodedValue;
21      }
22  
23      @Test
24      public void shouldEncode() throws CborException {
25          ByteArrayOutputStream byteOutputStream = new ByteArrayOutputStream();
26          CborEncoder encoder = new CborEncoder(byteOutputStream);
27          encoder.encode(new HalfPrecisionFloat(value));
28          Assert.assertArrayEquals(encodedValue, byteOutputStream.toByteArray());
29      }
30  
31      @Test
32      public void shouldDecode() throws CborException {
33          InputStream inputStream = new ByteArrayInputStream(encodedValue);
34          CborDecoder decoder = new CborDecoder(inputStream);
35          DataItem dataItem = decoder.decodeNext();
36          Assert.assertTrue(dataItem instanceof HalfPrecisionFloat);
37          HalfPrecisionFloat halfPrecisionFloat = (HalfPrecisionFloat) dataItem;
38          Assert.assertEquals(value, halfPrecisionFloat.getValue(), 0);
39      }
40  
41  }
