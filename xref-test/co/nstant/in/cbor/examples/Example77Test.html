3   import java.io.ByteArrayInputStream;
4   import java.io.ByteArrayOutputStream;
5   import java.io.InputStream;
6   import java.util.List;
7   
8   import org.junit.Assert;
9   import org.junit.Test;
10  
11  import co.nstant.in.cbor.CborBuilder;
12  import co.nstant.in.cbor.CborDecoder;
13  import co.nstant.in.cbor.CborEncoder;
14  import co.nstant.in.cbor.CborException;
15  import co.nstant.in.cbor.model.DataItem;
16  
17  /**
18   * [1, [2, 3], [_ 4,5]] -> 0x83 01 82 02 03 9f 04 05 ff
19   */
20  public class Example77Test {
21  
22  	private static final List<DataItem> VALUE = new CborBuilder().addArray()
23  			.add(1).addArray().add(2).add(3).end().startArray().add(4).add(5)
24  			.end().end().build();
25  
26  	private static final byte[] ENCODED_VALUE = new byte[] { (byte) 0x83, 0x01,
27  			(byte) 0x82, 0x02, 0x03, (byte) 0x9f, 0x04, 0x05, (byte) 0xff };
28  
29  	@Test
30  	public void shouldEncode() throws CborException {
31  		ByteArrayOutputStream byteOutputStream = new ByteArrayOutputStream();
32  		CborEncoder encoder = new CborEncoder(byteOutputStream);
33  		encoder.encode(VALUE);
34  		Assert.assertArrayEquals(ENCODED_VALUE, byteOutputStream.toByteArray());
35  	}
36  
37  	@Test
38  	public void shouldDecode() throws CborException {
39  		InputStream inputStream = new ByteArrayInputStream(ENCODED_VALUE);
40  		CborDecoder decoder = new CborDecoder(inputStream);
41  		List<DataItem> dataItems = decoder.decode();
42  		Assert.assertArrayEquals(VALUE.toArray(), dataItems.toArray());
43  	}
44  
45  }
