11  import co.nstant.in.cbor.CborBuilder;
12  import co.nstant.in.cbor.CborDecoder;
13  import co.nstant.in.cbor.CborEncoder;
14  import co.nstant.in.cbor.CborException;

18  
19  /**
20   * [_ 1, [2, 3], [_ 4, 5]] -> 0x9f018202039f0405ffff
21   */
22  public class Example75Test {
23  
24      private static final List<DataItem> VALUE = new CborBuilder()
25                      .add(new Array().setChunked(true))
26                      .add(1)
27                      .addArray().add(2).add(3).end()
28                      .add(new Array().setChunked(true))
29                      .add(4)
30                      .add(5)
31                      .add(SimpleValue.BREAK)
32                      .add(SimpleValue.BREAK)
33                      .build();
34  
35      private static final byte[] ENCODED_VALUE = new byte[] {
36                      (byte) 0x9f, 0x01, (byte) 0x82, 0x02, 0x03, (byte) 0x9f,
37                      0x04, 0x05, (byte) 0xff, (byte) 0xff };
38  
39      @Test
40      public void shouldEncode() throws CborException {
41          ByteArrayOutputStream byteOutputStream = new ByteArrayOutputStream();
42          CborEncoder encoder = new CborEncoder(byteOutputStream);
43          encoder.encode(VALUE);
44          Assert.assertArrayEquals(ENCODED_VALUE, byteOutputStream.toByteArray());
45      }
46  
47      @Test
48      public void shouldDecode() throws CborException {
49          InputStream inputStream = new ByteArrayInputStream(ENCODED_VALUE);
50          CborDecoder decoder = new CborDecoder(inputStream);
51          decoder.setAutoDecodeInfinitiveArrays(false);
52          List<DataItem> dataItems = decoder.decode();
53          Assert.assertArrayEquals(VALUE.toArray(), dataItems.toArray());
54      }
55  
56  }
