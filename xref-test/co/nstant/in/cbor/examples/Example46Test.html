3   import java.io.ByteArrayInputStream;
4   import java.io.ByteArrayOutputStream;
5   import java.io.InputStream;

7   import org.junit.Assert;
8   import org.junit.Test;
9   
10  import co.nstant.in.cbor.CborDecoder;
11  import co.nstant.in.cbor.CborEncoder;
12  import co.nstant.in.cbor.CborException;
13  import co.nstant.in.cbor.model.DataItem;
14  import co.nstant.in.cbor.model.SimpleValue;
15  
16  /**
17   * simple(24) -> 0xf818
18   */
19  public class Example46Test {
20  
21      private static final SimpleValue VALUE = new SimpleValue(24);
22      private static final byte[] ENCODED_VALUE = new byte[] { (byte) 0xf8, 0x18 };
23  
24      @Test
25      public void shouldEncode() throws CborException {
26          ByteArrayOutputStream byteOutputStream = new ByteArrayOutputStream();
27          CborEncoder encoder = new CborEncoder(byteOutputStream);
28          encoder.encode(VALUE);
29          Assert.assertArrayEquals(ENCODED_VALUE, byteOutputStream.toByteArray());
30      }
31  
32      @Test
33      public void shouldDecode() throws CborException {
34          InputStream inputStream = new ByteArrayInputStream(ENCODED_VALUE);
35          CborDecoder decoder = new CborDecoder(inputStream);
36          DataItem dataItem = decoder.decodeNext();
37          Assert.assertTrue(dataItem instanceof SimpleValue);
38          SimpleValue simpleValue = (SimpleValue) dataItem;
39          Assert.assertEquals(VALUE, simpleValue);
40      }
41  
42  }
