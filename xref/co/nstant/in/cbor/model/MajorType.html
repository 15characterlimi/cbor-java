3   public enum MajorType {
4   
5   	/**
6   	 * Major type 0: an unsigned integer. The 5-bit additional information is
7   	 * either the integer itself (for additional information values 0 through
8   	 * 23), or the length of additional data. Additional information 24 means
9   	 * the value is represented in an additional uint8_t, 25 means a uint16_t,
10  	 * 26 means a uint32_t, and 27 means a uint64_t. For example, the integer 10
11  	 * is denoted as the one byte 0b000_01010 (major type 0, additional
12  	 * information 10). The integer 500 would be 0b000_11001 (major type 0,
13  	 * additional information 25) followed by the two bytes 0x01f4, which is 500
14  	 * in decimal.
15  	 */
16  	UNSIGNED_INTEGER(0),
17  
18  	/**
19  	 * Major type 1: a negative integer. The encoding follows the rules for
20  	 * unsigned integers (major type 0), except that the value is then -1 minus
21  	 * the encoded unsigned integer. For example, the integer -500 would be
22  	 * 0b001_11001 (major type 1, additional information 25) followed by the two
23  	 * bytes 0x01f3, which is 499 in decimal.
24  	 */
25  	NEGATIVE_INTEGER(1),
26  
27  	/**
28  	 * Major type 2: a byte string. The string's length in bytes is represented
29  	 * following the rules for positive integers (major type 0). For example, a
30  	 * byte string whose length is 5 would have an initial byte of 0b010_00101
31  	 * (major type 2, additional information 5 for the length), followed by 5
32  	 * bytes of binary content. A byte string whose length is 500 would have 3
33  	 * initial bytes of 0b010_11001 (major type 2, additional information 25 to
34  	 * indicate a two-byte length) followed by the two bytes 0x01f4 for a length
35  	 * of 500, followed by 500 bytes of binary content.
36  	 */
37  	BYTE_STRING(2),
38  
39  	/**
40  	 * Major type 3: string of Unicode characters that is encoded as UTF-8
41  	 * [RFC3629]. The format of this type is identical to that of byte strings
42  	 * (major type 2), that is, as with major type 2, the length gives the
43  	 * number of bytes. This type is provided for systems that need to interpret
44  	 * or display human-readable text. In contrast to formats such as JSON, the
45  	 * Unicode characters in this type are never escaped. Thus, a newline
46  	 * character (U+000A) is always represented in a string as the byte 0x0a,
47  	 * and never as the bytes 0x5c6e (the characters "\" and "n") or as
48  	 * 0x5c7530303061 (the characters "\", "u", "0", "0", "0", and "a").
49  	 */
50  	UNICODE_STRING(3),
51  
52  	/**
53  	 * Major type 4: an array of data items. Arrays are also called lists,
54  	 * sequences, or tuples. The array's length follows the rules for byte
55  	 * strings (major type 2), except that the length denotes the number of data
56  	 * items, not the length in bytes that the array takes up. Items in an array
57  	 * do not need to all be of the same type. For example, an array that
58  	 * contains 10 items of any type would have an initial byte of 0b100_01010
59  	 * (major type of 4, additional information of 10 for the length) followed
60  	 * by the 10 remaining items.
61  	 */
62  	ARRAY(4),
63  
64  	/**
65  	 * Major type 5: a map of pairs of data items. Maps are also called tables,
66  	 * dictionaries, hashes, or objects (in JSON). A map is comprised of pairs
67  	 * of data items, the even-numbered ones serving as keys and the following
68  	 * odd-numbered ones serving as values for the key that comes immediately
69  	 * before it. The map's length follows the rules for byte strings (major
70  	 * type 2), except that the length denotes the number of pairs, not the
71  	 * length in bytes that the map takes up. For example, a map that contains 9
72  	 * pairs would have an initial byte of 0b101_01001 (major type of 5,
73  	 * additional information of 9 for the number of pairs) followed by the 18
74  	 * remaining items. The first item is the first key, the second item is the
75  	 * first value, the third item is the second key, and so on.
76  	 */
77  	MAP(5),
78  
79  	/**
80  	 * Major type 6: optional semantic tagging of other major types. See Section
81  	 * 2.4.
82  	 */
83  	TAG(6),
84  
85  	/**
86  	 * Major type 7: floating point numbers and simple data types that need no
87  	 * content, as well as the "break" stop code. See Section 2.3.
88  	 */
89  	SPECIAL(7);
90  
91  	private final int value;
92  
93  	private MajorType(int value) {
94  		this.value = value;
95  	}
96  
97  	public int getValue() {
98  		return value;
99  	}
100 
101 	public static MajorType ofByte(int b) {
102 		switch (b >> 5) {
103 		case 0:
104 			return UNSIGNED_INTEGER;
105 		case 1:
106 			return NEGATIVE_INTEGER;
107 		case 2:
108 			return BYTE_STRING;
109 		case 3:
110 			return UNICODE_STRING;
111 		case 4:
112 			return ARRAY;
113 		case 5:
114 			return MAP;
115 		case 6:
116 			return TAG;
117 		case 7:
118 			return SPECIAL;
119 		default:
120 			return null;
121 		}
122 	}
123 
124 }
