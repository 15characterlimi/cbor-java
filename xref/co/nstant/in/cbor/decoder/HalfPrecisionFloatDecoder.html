1   package co.nstant.in.cbor.decoder;

3   import java.io.InputStream;
4   
5   import co.nstant.in.cbor.CborDecoder;
6   import co.nstant.in.cbor.CborException;
7   import co.nstant.in.cbor.model.HalfPrecisionFloat;
8   
9   public class HalfPrecisionFloatDecoder extends
10                  AbstractDecoder<HalfPrecisionFloat> {
11  
12      public HalfPrecisionFloatDecoder(CborDecoder decoder,
13                      InputStream inputStream) {
14          super(decoder, inputStream);
15      }
16  
17      @Override
18      public HalfPrecisionFloat decode(int initialByte) throws CborException {
19          int bits = nextSymbol() << 8 | nextSymbol();
20          return new HalfPrecisionFloat(toFloat(bits));
21      }
22  
23      /**
24       * @see http://stackoverflow.com/a/5684578
25       */
26      private static float toFloat(int bits) {
27          int s = (bits & 0x8000) >> 15;
28          int e = (bits & 0x7C00) >> 10;
29          int f = bits & 0x03FF;
30  
31          if (e == 0) {
32              return (float) ((s != 0 ? -1 : 1) * Math.pow(2, -14) * (f / Math
33                              .pow(2, 10)));
34          } else if (e == 0x1F) {
35              return f != 0 ? Float.NaN
36                              : (s != 0 ? -1 : 1) * Float.POSITIVE_INFINITY;
37          }
38  
39          return (float) ((s != 0 ? -1 : 1) * Math.pow(2, e - 15) * (1 + f / Math
40                          .pow(2, 10)));
41      }
42  
43  }
