28  /**
29   * Encoder for the CBOR format based.
30   */
31  public class CborEncoder {
32  
33  	private final UnsignedIntegerEncoder unsignedIntegerEncoder;
34  	private final NegativeIntegerEncoder negativeIntegerEncoder;
35  	private final ByteStringEncoder byteStringEncoder;
36  	private final UnicodeStringEncoder unicodeStringEncoder;
37  	private final ArrayEncoder arrayEncoder;
38  	private final MapEncoder mapEncoder;
39  	private final TagEncoder tagEncoder;
40  	private final SpecialEncoder specialEncoder;
41  	private final RationalNumberEncoder rationalNumberEncoder;
42  
43  	/**
44  	 * Initialize a new encoder which writes the binary encoded data to an
45  	 * {@link OutputStream}.
46  	 */
47  	public CborEncoder(OutputStream outputStream) {
48  		Objects.requireNonNull(outputStream);
49  		unsignedIntegerEncoder = new UnsignedIntegerEncoder(this, outputStream);
50  		negativeIntegerEncoder = new NegativeIntegerEncoder(this, outputStream);
51  		byteStringEncoder = new ByteStringEncoder(this, outputStream);
52  		unicodeStringEncoder = new UnicodeStringEncoder(this, outputStream);
53  		arrayEncoder = new ArrayEncoder(this, outputStream);
54  		mapEncoder = new MapEncoder(this, outputStream);
55  		tagEncoder = new TagEncoder(this, outputStream);
56  		specialEncoder = new SpecialEncoder(this, outputStream);
57  		rationalNumberEncoder = new RationalNumberEncoder(this, outputStream);
58  	}
59  
60  	/**
61  	 * Encode a list of {@link DataItem}s, also known as a stream.
62  	 * 
63  	 * @param dataItems
64  	 *            a list of {@link DataItem}s
65  	 * @throws CborException
66  	 *             if the {@link DataItem}s could not be encoded or there was an
67  	 *             problem with the {@link OutputStream}.
68  	 */
69  	public void encode(List<DataItem> dataItems) throws CborException {
70  		for (DataItem dataItem : dataItems) {
71  			encode(dataItem);
72  		}
73  	}
74  
75  	/**
76  	 * Encode a single {@link DataItem}.
77  	 * 
78  	 * @param dataItem
79  	 *            the {@link DataItem} to encode. If null, encoder encodes a
80  	 *            {@link SimpleValue} NULL value.
81  	 * @throws CborException
82  	 *             if {@link DataItem} could not be encoded or there was an
83  	 *             problem with the {@link OutputStream}.
84  	 */
85  	public void encode(DataItem dataItem) throws CborException {
86  		if (dataItem == null) {
87  			dataItem = SimpleValue.NULL;
88  		}
89  		switch (dataItem.getMajorType()) {
90  		case UNSIGNED_INTEGER:
91  			unsignedIntegerEncoder.encode((UnsignedInteger) dataItem);
92  			break;
93  		case NEGATIVE_INTEGER:
94  			negativeIntegerEncoder.encode((NegativeInteger) dataItem);
95  			break;
96  		case BYTE_STRING:
97  			byteStringEncoder.encode((ByteString) dataItem);
98  			break;
99  		case UNICODE_STRING:
100 			unicodeStringEncoder.encode((UnicodeString) dataItem);
101 			break;
102 		case ARRAY:
103 			arrayEncoder.encode((Array) dataItem);
104 			break;
105 		case MAP:
106 			mapEncoder.encode((Map) dataItem);
107 			break;
108 		case SPECIAL:
109 			specialEncoder.encode((Special) dataItem);
110 			break;
111 		case TAG:
112 			tagEncoder.encode((Tag) dataItem);
113 			break;
114 		default:
115 			throw new IllegalArgumentException("Unknown major type");
116 		}
117 	}
118 
119 	public void encode(RationalNumber rationalNumber) throws CborException {
120 		rationalNumberEncoder.encode(rationalNumber);
121 	}
122 
123 }
