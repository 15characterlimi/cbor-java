17      @Override
18      public void encode(HalfPrecisionFloat dataItem) throws CborException {
19          write((7 << 5) | 25);
20          int bits = fromFloat(dataItem.getValue());
21          write((bits >> 8) & 0xFF);
22          write((bits >> 0) & 0xFF);
23      }

29      // returns all higher 16 bits as 0 for all results
30      public static int fromFloat(float fval) {
31          int fbits = Float.floatToIntBits(fval);
32          int sign = fbits >>> 16 & 0x8000; // sign only
33          int val = 0x1000 + fbits & 0x7fffffff; // rounded value
34  
35          if (val >= 0x47800000) // might be or become NaN/Inf
36          { // avoid Inf due to rounding
37              if ((fbits & 0x7fffffff) >= 0x47800000)
38              { // is or must become NaN/Inf
39                  if (val < 0x7f800000) // was value but too large
40                      return sign | 0x7c00; // make it +/-Inf
41                  return sign | 0x7c00 | // remains +/-Inf or NaN
42                                  (fbits & 0x007fffff) >>> 13; // keep NaN (and
43                                                               // Inf) bits
44              }
45              return sign | 0x7bff; // unrounded not quite Inf
46          }
47          if (val >= 0x38800000) // remains normalized value
48              return sign | val - 0x38000000 >>> 13; // exp - 127 + 15
49          if (val < 0x33000000) // too small for subnormal
50              return sign; // becomes +/-0
51          val = (fbits & 0x7fffffff) >>> 23; // tmp exp for subnormal calc
52          return sign | ((fbits & 0x7fffff | 0x800000) // add subnormal bit
53                          + (0x800000 >>> val - 102) // round depending on cut off
54          >>> 126 - val); // div by 2^(1-(exp-127+15)) and >> 13 | exp=0
55      }
