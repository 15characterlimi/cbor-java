3   import java.io.OutputStream;
4   
5   import co.nstant.in.cbor.CborEncoder;
6   import co.nstant.in.cbor.CborException;
7   import co.nstant.in.cbor.model.HalfPrecisionFloat;
8   
9   public class HalfPrecisionFloatEncoder extends AbstractEncoder<HalfPrecisionFloat> {
10  
11  	public HalfPrecisionFloatEncoder(CborEncoder encoder, OutputStream outputStream) {
12  		super(encoder, outputStream);
13  	}
14  
15  	@Override
16  	public void encode(HalfPrecisionFloat dataItem) throws CborException {
17  		write((7 << 5) | 25);
18  		int bits = fromFloat(dataItem.getValue());
19  		write((bits >> 8) & 0xFF);
20  		write((bits >> 0) & 0xFF);
21  	}
22  
23  	/**
24  	 * @see http://stackoverflow.com/a/6162687
25  	 */
26  
27  	// returns all higher 16 bits as 0 for all results
28  	public static int fromFloat(float fval) {
29  		int fbits = Float.floatToIntBits(fval);
30  		int sign = fbits >>> 16 & 0x8000; // sign only
31  		int val = 0x1000 + fbits & 0x7fffffff; // rounded value
32  
33  		if (val >= 0x47800000) // might be or become NaN/Inf
34  		{ // avoid Inf due to rounding
35  			if ((fbits & 0x7fffffff) >= 0x47800000) { // is or must become
36  														// NaN/Inf
37  				if (val < 0x7f800000) {// was value but too large
38  					return sign | 0x7c00; // make it +/-Inf
39  				}
40  				return sign | 0x7c00 | // remains +/-Inf or NaN
41  						(fbits & 0x007fffff) >>> 13; // keep NaN (and
42  														// Inf) bits
43  			}
44  			return sign | 0x7bff; // unrounded not quite Inf
45  		}
46  		if (val >= 0x38800000) { // remains normalized value
47  			return sign | val - 0x38000000 >>> 13; // exp - 127 + 15
48  		}
49  		if (val < 0x33000000) { // too small for subnormal
50  			return sign; // becomes +/-0
51  		}
52  		val = (fbits & 0x7fffffff) >>> 23; // tmp exp for subnormal calc
53  		return sign | ((fbits & 0x7fffff | 0x800000) // add subnormal bit
54  				+ (0x800000 >>> val - 102) // round depending on cut off
55  		>>> 126 - val); // div by 2^(1-(exp-127+15)) and >> 13 | exp=0
56  	}
57  
58  }
