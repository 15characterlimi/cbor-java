3   import java.io.OutputStream;
4   import java.nio.charset.Charset;
5   
6   import co.nstant.in.cbor.CborEncoder;
7   import co.nstant.in.cbor.CborException;
8   import co.nstant.in.cbor.model.MajorType;
9   import co.nstant.in.cbor.model.SimpleValue;
10  import co.nstant.in.cbor.model.UnicodeString;
11  
12  public class UnicodeStringEncoder extends AbstractEncoder<UnicodeString> {
13  
14      private static final Charset UTF8 = Charset.forName("UTF8");
15  
16      public UnicodeStringEncoder(CborEncoder encoder, OutputStream outputStream) {
17          super(encoder, outputStream);
18      }
19  
20      @Override
21      public void encode(UnicodeString dataItem) throws CborException {
22          String string = dataItem.getString();
23          if (dataItem.isChunked()) {
24              encodeTypeChunked(MajorType.UNICODE_STRING);
25              if (string != null) {
26                  encode(new UnicodeString(string));
27              }
28          } else if (string == null) {
29              encoder.encode(SimpleValue.NULL);
30          } else {
31              byte[] bytes = string.getBytes(UTF8);
32              encodeTypeAndLength(MajorType.UNICODE_STRING, bytes.length);
33              write(bytes);
34          }
35      }
36  
37  }
