1   package co.nstant.in.cbor.encoder;
2   
3   import java.io.OutputStream;
4   
5   import co.nstant.in.cbor.CborEncoder;
6   import co.nstant.in.cbor.CborException;
7   import co.nstant.in.cbor.model.DoublePrecisionFloat;
8   import co.nstant.in.cbor.model.HalfPrecisionFloat;
9   import co.nstant.in.cbor.model.SimpleValue;
10  import co.nstant.in.cbor.model.SimpleValueType;
11  import co.nstant.in.cbor.model.SinglePrecisionFloat;
12  import co.nstant.in.cbor.model.Special;
13  
14  public class SpecialEncoder extends AbstractEncoder<Special> {
15  
16  	private final HalfPrecisionFloatEncoder halfPrecisionFloatEncoder;
17  	private final SinglePrecisionFloatEncoder singlePrecisionFloatEncoder;
18  	private final DoublePrecisionFloatEncoder doublePrecisionFloatEncoder;
19  
20  	public SpecialEncoder(CborEncoder encoder, OutputStream outputStream) {
21  		super(encoder, outputStream);
22  		halfPrecisionFloatEncoder = new HalfPrecisionFloatEncoder(encoder, outputStream);
23  		singlePrecisionFloatEncoder = new SinglePrecisionFloatEncoder(encoder, outputStream);
24  		doublePrecisionFloatEncoder = new DoublePrecisionFloatEncoder(encoder, outputStream);
25  	}
26  
27  	@Override
28  	public void encode(Special dataItem) throws CborException {
29  		switch (dataItem.getSpecialType()) {
30  		case BREAK:
31  			write((7 << 5) | 31);
32  			break;
33  		case SIMPLE_VALUE:
34  			SimpleValue simpleValue = (SimpleValue) dataItem;
35  			switch (simpleValue.getSimpleValueType()) {
36  			case FALSE:
37  			case NULL:
38  			case TRUE:
39  			case UNDEFINED:
40  				SimpleValueType type = simpleValue.getSimpleValueType();
41  				write((7 << 5) | type.getValue());
42  				break;
43  			case UNALLOCATED:
44  				write((7 << 5) | simpleValue.getValue());
45  				break;
46  			case RESERVED:
47  			default:
48  				throw new CborException("Not implemented");
49  			}
50  			break;
51  		case UNALLOCATED:
52  			throw new CborException("Unallocated special type");
53  		case IEEE_754_HALF_PRECISION_FLOAT:
54  			if (!(dataItem instanceof HalfPrecisionFloat)) {
55  				throw new CborException("Wrong data item type");
56  			}
57  			halfPrecisionFloatEncoder.encode((HalfPrecisionFloat) dataItem);
58  			break;
59  		case IEEE_754_SINGLE_PRECISION_FLOAT:
60  			if (!(dataItem instanceof SinglePrecisionFloat)) {
61  				throw new CborException("Wrong data item type");
62  			}
63  			singlePrecisionFloatEncoder.encode((SinglePrecisionFloat) dataItem);
64  			break;
65  		case IEEE_754_DOUBLE_PRECISION_FLOAT:
66  			if (!(dataItem instanceof DoublePrecisionFloat)) {
67  				throw new CborException("Wrong data item type");
68  			}
69  			doublePrecisionFloatEncoder.encode((DoublePrecisionFloat) dataItem);
70  			break;
71  		case SIMPLE_VALUE_NEXT_BYTE:
72  			if (!(dataItem instanceof SimpleValue)) {
73  				throw new CborException("Wrong data item type");
74  			}
75  			SimpleValue simpleValueNextByte = (SimpleValue) dataItem;
76  			write((7 << 5) | 24);
77  			write(simpleValueNextByte.getValue());
78  			break;
79  		default:
80  			throw new CborException("Not implemented");
81  		}
82  	}
83  
84  }
